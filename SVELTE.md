# Installation de Node.js et npm

https://deb.nodesource.com/

# Initialisation de l'application SvelteKit

  npm create svelte@lastest mon_app
  cd mon_app
  npm install

# Initialisation d'une app Svelte, sans SvelteKit

  npm create vite@latest
  cd mon_app
  npm install
  
  npm run build
